 Typing scene system
function typeScene(lines, callback) {
  container.innerHTML = div id='text'div;
  let el = document.getElementById(text);
  let i = 0;

  function showNextLine() {
    if (i  lines.length) {
      let p = document.createElement(p);
      p.className = text;
      p.innerHTML = lines[i];
      el.appendChild(p);
      i++;
      setTimeout(showNextLine, 2800);  smoother pacing
    } else {
      if (callback) setTimeout(callback, 1500);
    }
  }

  showNextLine();
}
